#include 
#include "hello.h"

int main(int argc, char **argv)
{
    char *msg;

    if (argc != 2) {
	fprintf(stderr, "Usage: %s message\n", argv[0]);
	exit(1);
    }

    msg = hello(argv[1]);
    if (!msg) {
	fprintf(stderr, "Error creating message!\n");
	exit(1);
    }

    printf("%s\n", msg);
    free(msg);
    
    return 0;
}
