#include 
#include "hello.h"

int main(int argc, char **argv)
{
    char *msg;
    int i;
    
    if (argc < 2) {
	fprintf(stderr, "Usage: %s message\n", argv[0]);
	exit(1);
    }

    for (i=1; i<argc; i++) {
    	msg = hello(argv[i]);
    	if (!msg) {
		fprintf(stderr, "Error creating message!\n");
		exit(1);
    	}

    	printf("%s\n", msg);
    	free(msg);
    }
    	
    return 0;
}
