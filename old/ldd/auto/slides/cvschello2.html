#include 
#include "hello.h"

int main(int argc, char **argv)
{
    char all[256];
    char *msg;
    int i;

    if (argc < 2) {
	fprintf(stderr, "Usage: %s message\n", argv[0]);
	exit(1);
    }

    all[0] = '\0';
    for (i=1; i<argc; i++) {
	strcat(all, argv[i]);
	strcat(all, ", ");
    }
    all[strlen(all)-2] = '\0';
    
    msg = hello(all);
    if (!msg) {
	fprintf(stderr, "Error creating message!\n");
	exit(1);
    }

    printf("%s\n", msg);
    free(msg);
 
    return 0;
}
